\begin{algorithm}
\caption{Euclid's Algorithm}\label{alg:euclid}
\begin{algorithmic}[1]
\Procedure{GCD }{$a,b$}
\State $r\gets a\bmod b$
\While{$r\not=0$}\Comment{only stop when r is zero}
\State $a\gets b$
\State $b\gets r$
\State $r\gets a\bmod b$
\EndWhile
\State \textbf{return} $b$\Comment{return gcd: b}
\EndProcedure
\end{algorithmic}
\end{algorithm}
